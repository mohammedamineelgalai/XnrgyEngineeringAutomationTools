Example
  • C#
using Autodesk.Connectivity.WebServices;
using Autodesk.Connectivity.WebServicesTools;
using Autodesk.DataManagement.Client.Framework.Vault.Currency.Entities;

class VaultSDKSample
{
    public static void AddNumberingSchemeSample(WebServiceManager serviceManager)
    {
        // Get the entity class ID we're working with
        string entityClassId = EntityClassIds.Items;

        // Get the default Numbering Provider for our Numbering Scheme
        NumProv defaultProvider = serviceManager.NumberingService.GetNumberingProviders()[0];

        // Add a numbering scheme that can be either "Part-######" or "Assembly-######"
        string schemeName = "TestScheme";
        bool toUpperCase = false;
        bool reuseNumbers = false;

        // create the predefined list field
        CodeWord partWord = new CodeWord();
        partWord.Code = "Part";
        CodeWord assemblyWord = new CodeWord();
        assemblyWord.Code = "Assembly";
        PredefListField listField = new PredefListField();
        listField.Name = "Type";
        listField.CodeArray = new CodeWord[] { partWord, assemblyWord };
        listField.DfltVal = partWord.Code;

        // create a delimiter field
        DelimField delimiterField = new DelimField();
        delimiterField.DelimVal = '-';

        // create an autodenerated field
        AutogenField autogenField = new AutogenField();
        autogenField.Name = "Number";
        autogenField.Len = 4;
        autogenField.From = 1;
        autogenField.To = 9999;
        autogenField.StepSize = 1;
        autogenField.ZeroPadding = true;

        NumSchmField[] fieldArray = new NumSchmField[] { listField, delimiterField, autogenField };

        serviceManager.NumberingService.AddNumberingScheme(entityClassId, schemeName, defaultProvider.SysName, fieldArray, toUpperCase, reuseNumbers);
    }
}
